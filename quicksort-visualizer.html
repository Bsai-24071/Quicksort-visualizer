function quickSort(arr, lo, hi):
if lo < hi:
p = partition(arr, lo, hi)
quickSort(arr, lo, p-1)
quickSort(arr, p+1, hi)
 
function partition(arr, lo, hi):
pivot = arr[hi] // chosen by strategy
i = lo - 1
for j = lo to hi - 1:
if arr[j] <= pivot:
i++
swap(arr[i], arr[j])
swap(arr[i+1], arr[hi])
return i + 1
